171  AffineTransform& rotate(float theta) {
172  const float c = std::cos(theta);
173  const float s = std::sin(theta);
174 
175  const float new00 = at(0, 0) * c + at(0, 1) * s;
176  const float new01 = at(0, 0) * -s + at(0, 1) * c;
177  const float new10 = at(1, 0) * c + at(1, 1) * s;
178  const float new11 = at(1, 0) * -s + at(1, 1) * c;
179 
180  at(0, 0) = new00;
181  at(0, 1) = new01;
182  at(1, 0) = new10;
183  at(1, 1) = new11;
184 
185  rotation += theta;
186 
187  return *this;
188  }

233  AffineTransform& concatenate(const AffineTransform& t){
234  const float new00 = constAt(0,0) * t.constAt(0,0) + constAt(0,1) * t.constAt(1,0);
235  const float new01 = constAt(0,0) * t.constAt(0,1) + constAt(0,1) * t.constAt(1,1);
236  const float new02 = constAt(0,0) * t.constAt(0,2) + constAt(0,1) * t.constAt(1,2) + constAt(0,2);
237  const float new10 = constAt(1,0) * t.constAt(0,0) + constAt(1,1) * t.constAt(1,0);
238  const float new11 = constAt(1,0) * t.constAt(0,1) + constAt(1,1) * t.constAt(1,1);
239  const float new12 = constAt(1,0) * t.constAt(0,2) + constAt(1,1) * t.constAt(1,2) + constAt(1,2);
240 
241  at(0,0) = new00;
242  at(0,1) = new01;
243  at(0,2) = new02;
244  at(1,0) = new10;
245  at(1,1) = new11;
246  at(1,2) = new12;
247  rotation += t.rotation;
248  return *this;
249  }

273  Point transform(Point in, Point* dst = nullptr) const{
274  Point temp;
275  Point* dstPtr = (dst == nullptr) ? &temp : dst;
276  dstPtr->x = (constAt(0, 0) * (in.x) + constAt(0, 1) * (in.y) + constAt(0, 2));
277  dstPtr->y = (constAt(1, 0) * (in.x) + constAt(1, 1) * (in.y) + constAt(1, 2));
278  return *dstPtr;
