182  AffineTransform& rotate(float theta) {
183  //6.28319 is a full circle in radians.
184  if(rotation + theta > 6.28319){//Loop back around. Avoid overflow.
185  theta = (rotation + theta) - 6.28319;
186  setRotation(0);
187  }
188  const float c = std::cos(theta);
189  const float s = std::sin(theta);
190 
191  const float new00 = at(0, 0) * c + at(0, 1) * s;
192  const float new01 = at(0, 0) * -s + at(0, 1) * c;
193  const float new10 = at(1, 0) * c + at(1, 1) * s;
194  const float new11 = at(1, 0) * -s + at(1, 1) * c;
195 
196  at(0, 0) = new00;
197  at(0, 1) = new01;
198  at(1, 0) = new10;
199  at(1, 1) = new11;
200 
201  rotation += theta;
202 
203  return *this;
204  }

527  void drawLine(Image& imgRef, int x1, int y1, int x2, int y2, Color c, unsigned int width){
528  if (x1 == x2 && y1 == y2) {
529  return;
530  }
531  const double xoffs = std::abs(x1 - x2);
532  const double yoffs = std::abs(y1 - y2);
533  const double woffs = width / 2.0;
534  const int xadjacent = yoffs * woffs / std::sqrt(std::pow(xoffs, 2) + std::pow(yoffs, 2));
535  const int yadjacent = xoffs * woffs / std::sqrt(std::pow(xoffs, 2) + std::pow(yoffs, 2));
536 
537  //Clockwise
538  cimg_library::CImg<int> points(4, 2);
539  points(0, 0) = x1 - xadjacent;
540  points(0, 1) = y1 + yadjacent;
541  points(1, 0) = x1 + xadjacent;
542  points(1, 1) = y1 - yadjacent;
543  points(2, 0) = x2 + xadjacent;
544  points(2, 1) = y2 - yadjacent;
545  points(3, 0) = x2 - xadjacent;
546  points(3, 1) = y2 + yadjacent;
547  imgRef.draw_polygon(points, c.rgbPtr());
548  }
