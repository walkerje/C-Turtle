212  AffineTransform& rotate(float theta) {
213  //6.28319 is a full circle in radians.
214  if(rotation + theta > 6.28319){//Loop back around. Avoid overflow.
215  theta = (rotation + theta) - 6.28319;
216  setRotation(0);
217  }
218  const float c = std::cos(theta);
219  const float s = std::sin(theta);
220 
221  const float new00 = at(0, 0) * c + at(0, 1) * s;
222  const float new01 = at(0, 0) * -s + at(0, 1) * c;
223  const float new10 = at(1, 0) * c + at(1, 1) * s;
224  const float new11 = at(1, 0) * -s + at(1, 1) * c;
225 
226  at(0, 0) = new00;
227  at(0, 1) = new01;
228  at(1, 0) = new10;
229  at(1, 1) = new11;
230 
231  rotation += theta;
232 
233  return *this;
234  }

290  const float new00 = constAt(0,0) * t.constAt(0,0) + constAt(0,1) * t.constAt(1,0);

346  Point transform(Point in, Point* dst = nullptr) const{
347  Point temp;
348  Point* dstPtr = (dst == nullptr) ? &temp : dst;
349  //Rounding seems to fix off-by-one issues in regards to rotation.
350  dstPtr->x = std::round(constAt(0, 0) * (in.x) + constAt(0, 1) * (in.y) + constAt(0, 2));
351  dstPtr->y = std::round(constAt(1, 0) * (in.x) + constAt(1, 1) * (in.y) + constAt(1, 2));
352  return *dstPtr;
353  }

612  void drawLine(Image& imgRef, int x1, int y1, int x2, int y2, Color c, unsigned int width){
613  if (x1 == x2 && y1 == y2) {
614  return;
615  }else if(width == 1){
616  //Just use the built-in bresenham line function
617  //to draw line with widths of 1.
618  imgRef.draw_line(x1,y1,x2,y2,c.rgbPtr());
619  return;
620  }
621  //We pretty much re-implement Bresenham's Line Algorithm here,
622  //however instead of blitting pixels at each spot we put circles,
623  //which matches the rounded thick lines present in the Python implementation.
624  //This also allows for variable width.
625 
626  const bool isSteep = (std::abs(y2 - y1) > std::abs(x2 - x1));
627  if(isSteep){
628  std::swap(x1, y1);
629  std::swap(x2, y2);
630  }
631 
632  if(x1 > x2){
633  std::swap(x1, x2);
634  std::swap(y1, y2);
635  }
636 
637  const int dx = x2 - x1;
638  const int dy = std::abs(y2 - y1);
639 
640  int err = dx / 2;
641  const int ystep = (y1 < y2) ? 1 : -1;
642  int y = y1;
643 
644  const int maxX = x2;
645 
646  const int radius = std::ceil(float(width) / 2.0f);
647 
648  for(int x = x1; x < maxX; x++){
649  if(isSteep)
650  imgRef.draw_circle(y, x, radius, c.rgbPtr());
651  else
652  imgRef.draw_circle(x, y, radius, c.rgbPtr());
653  err -= dy;
654  if(err < 0){
655  y += ystep;
656  err += dx;
657  }
658  }
659  }

671  void Circle::draw(const AffineTransform& t, Image& imgRef, Color c, int outlineWidth, Color outlineColor) {
672  if(steps <= 0)
673  return;//no step check
674  cimg::CImg<int> passPts(steps, 2);
675 
676  for(int i = 0; i < steps; i++){
677  Point p;
678  p.x = int(radius * std::cos(i * (2*M_PI) / steps));
679  p.y = int(radius * std::sin(i * (2*M_PI) / steps));
680  Point tPoint = t(p);
681  passPts(i, 0) = tPoint.x;
682  passPts(i, 1) = tPoint.y;
683  }
684 
685  imgRef.draw_polygon(passPts, c.rgbPtr());
686 
687  if(outlineWidth > 0){//draw outline using previously generated points.
688  //LineLoop impl
689  for(int i = 1; i < steps; i++){
690  drawLine(imgRef, passPts(i-1, 0), passPts(i-1, 1), passPts(i, 0), passPts(i, 1), outlineColor, outlineWidth);
691  }
692  //draw last line between first and last
693  drawLine(imgRef, passPts(steps-1, 0), passPts(steps-1, 1), passPts(0, 0), passPts(0, 1), outlineColor, outlineWidth);
694  }
