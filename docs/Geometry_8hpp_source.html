170  AffineTransform& rotate(float theta) {
171  const float c = std::cos(theta);

175  const float new01 = at(0, 0) * -s + at(0, 1) * c;
176  const float new10 = at(1, 0) * c + at(1, 1) * s;
177  const float new11 = at(1, 0) * -s + at(1, 1) * c;

232  AffineTransform& concatenate(const AffineTransform& t){
233  const float new00 = constAt(0,0) * t.constAt(0,0) + constAt(0,1) * t.constAt(1,0);
234  const float new01 = constAt(0,0) * t.constAt(0,1) + constAt(0,1) * t.constAt(1,1);

236  const float new10 = constAt(1,0) * t.constAt(0,0) + constAt(1,1) * t.constAt(1,0);
237  const float new11 = constAt(1,0) * t.constAt(0,1) + constAt(1,1) * t.constAt(1,1);
238  const float new12 = constAt(1,0) * t.constAt(0,2) + constAt(1,1) * t.constAt(1,2) + constAt(1,2);
239

241  at(0,1) = new01;
242  at(0,2) = new02;
243  at(1,0) = new10;
244  at(1,1) = new11;
245  at(1,2) = new12;
246  rotation += t.rotation;
