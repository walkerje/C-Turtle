107  inline constexpr color_int_t resolveColorInt(uint8_t r, uint8_t g, uint8_t b) {
108  return ((r & 0xFF) << 16) + ((g & 0xFF) << 8) + (b & 0xFF);
109  }

116  inline void resolveColorComp(color_int_t pack, uint8_t& r, uint8_t& g, uint8_t& b) {
117  r = (pack & 0x00FF0000) >> 16; //Red
118  g = (pack & 0x0000FF00) >> 8; //Green
119  b = (pack & 0x000000FF); // >> 0; //Blue
120  }
